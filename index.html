Prepack оптимизирует исходники js.
Зачем код выполнять в run-time, если это можно сделать заранее?
Prepack оптимизирует исходники js. Prepack находит участки кода, которые можно оптимизировать и заменяет оптимизирует. Это избавляет от вспомогательных вычеслений в рантайме.

Пример#1

(function () {
  var self = this;
  ['A', 'B', 42].forEach(function(x) {
    var name = '_' + x.toString()[0].toLowerCase();
    var y = parseInt(x);
    self[name] = y ? y : x;
  });
})();
(function () {
  _a = "A";
  _b = "B";
  _4 = 42;
})();

Пример#2 Fibonacci

(function () {
  function fibonacci(x) {
    return x <= 1 ? x : fibonacci(x - 1) + fibonacci(x - 2);
  }
  global.x = fibonacci(23);
})();
(function () {
  x = 28657;
})();
